\begin{frame}{Adjoint Method}
    \begin{itemize}
        \item Airfoil optimization
        \item Require primal and adjoint simulation
        \item Complexity is independent of the number of parameters (eg control points on the airfoil)
    \end{itemize}

\end{frame}
